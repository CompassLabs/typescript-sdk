## tokenPrice

Retrieves the price of a token in USD.

### Example Usage

<!-- UsageSnippet language="typescript" operationID="v1_token_price" method="get" path="/v1/token/price" -->
```typescript

const compassApiSDK = new CompassApiSDK({
  apiKeyAuth: "<YOUR_API_KEY_HERE>",
});

  const result = await compassApiSDK.token.tokenPrice({
    token: "USDC",
  });

}
```

## tokenTransfer

Sends native ETH or ERC20 tokens from the sender's address to another address.

### Example Usage

<!-- UsageSnippet language="typescript" operationID="v1_token_transfer" method="post" path="/v1/token/transfer" -->
```typescript
import { CompassApiSDK } from "@compass-labs/api-sdk";

const compassApiSDK = new CompassApiSDK({
  apiKeyAuth: "<YOUR_API_KEY_HERE>",
});

async function run() {
  const result = await compassApiSDK.token.tokenTransfer({
    to: "0x68b3465833fb72A70ecDF485E0e4C7bD8665Fc44",
    token: "USDC",
    amount: 1.5,
    chain: "ethereum",
    sender: "0x29F20a192328eF1aD35e1564aBFf4Be9C5ce5f7B",
  });

  console.log(result);
}

run();
```
